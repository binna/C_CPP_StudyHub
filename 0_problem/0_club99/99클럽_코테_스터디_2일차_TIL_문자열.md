* 제출한 코드 <br>
▼ 실패한 코드
```
#include <string>
#include <vector>
using namespace std;
    
int solution(string t, string p)
{
    int answer = 0;
    int pSize = p.length();
    int pNum = stoi(p);

    for (int i = 0; i < t.length() - pSize + 1; i++)
    {
        int temp = stoi(t.substr(i, pSize));

        if (temp <= pNum)
        {
            answer += 1;
        }
    }

    return answer;
}
```

▼ 성공한 코드
```
#include <string>
using namespace std;
  
int solution(string t, string p) 
{
    int answer = 0;
    int pSize = p.length();
    long pNum = stol(p);
        
    for (int i = 0; i < t.length() - pSize + 1; i++)
    {
        long temp = stol(t.substr(i, pSize));
        
        if (temp <= pNum)
        {
            answer += 1;
        }
    }
        
    return answer;
}
```
