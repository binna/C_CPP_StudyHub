* 제출한 코드 <br>

```
#include <string>
#include <iostream>
using namespace std;

bool solution(string s)
{
    int pCnt = 0;
    int yCnt = 0;

    for (int i = 0; i < s.size(); i++)
    {
        if (s[i] == 'y' || s[i] == 'Y')
            yCnt += 1;
        else if (s[i] == 'p' || s[i] == 'P')
            pCnt += 1;
    }

    return (yCnt == pCnt);
}
```

```
#include <string>
#include <iostream>
using namespace std;

bool solution(string s)
{
    int pCnt = 0;
    int yCnt = 0;

    for (char word : s)
    {
        if (word == 'y' || word == 'Y')
            yCnt += 1;
        else if (word == 'p' || word == 'P')
            pCnt += 1;
    }

    return (yCnt == pCnt);
}
```
